CREATE TABLE T_AUDIT_GOODS (
    AUDITTIME TIMESTAMP NOT NULL PRIMARY KEY,
    NAME1 VARCHAR(255),
    NAME2 VARCHAR(255),
    COST1 D_MONEY NOT NULL,
    COST2 D_MONEY NOT NULL,
    GOODNUMBER1 INTEGER,
    GOODNUMBER2 INTEGER,
    STORAGENUMBER1 INTEGER,
    STORAGENUMBER2 INTEGER
);

CREATE TABLE TMP_CHECK_GOODS (
	GOODIDX INTEGER NOT NULL PRIMARY KEY,
	NUMBER INTEGER DEFAULT 0,
	CHECKED INTEGER DEFAULT 0,
	REASON VARCHAR(255),
	DESP VARCHAR(1024)
);
